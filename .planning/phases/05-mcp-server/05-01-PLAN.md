1. Create `src/mcp/server.ts`:
   ```typescript
   import { McpServer } from '@modelcontextprotocol/sdk/server/mcp.js';
   import { StdioServerTransport } from '@modelcontextprotocol/sdk/server/stdio.js';
   import { DiagramService } from '../diagram/service.js';
   import { log } from '../utils/logger.js';

   export function createMcpServer(service: DiagramService): McpServer {
     const server = new McpServer({
       name: 'smartb-diagrams',
       version: '0.1.0',
     });
     // Tools and resources registered in subsequent plans
     return server;
   }

   export async function startMcpServer(projectDir: string): Promise<void> {
     const { resolve } = await import('node:path');
     const resolvedDir = resolve(projectDir);
     const service = new DiagramService(resolvedDir);
     const server = createMcpServer(service);
     const transport = new StdioServerTransport();
     await server.connect(transport);
     log.info('MCP server running on stdio');
   }
   ```

   CRITICAL: The logger uses `console.error()` for all output (confirmed in utils/logger.ts). This is correct for MCP -- no stdout writes are allowed. Never use `console.log()` anywhere in MCP code paths.

2. Add `smartb mcp` CLI command to `src/cli.ts`:
   ```typescript
   program
     .command('mcp')
     .description('Start the MCP server for AI tool integration (stdio transport)')
     .option('-d, --dir <path>', 'project directory', '.')
     .action(async (options: { dir: string }) => {
       const { startMcpServer } = await import('./mcp/server.js');
       await startMcpServer(options.dir);
     });
   ```

   Place this BEFORE `program.parse()`. Follow the same dynamic import pattern used by the `serve` command.
